const SOURCE = "lista.m3u"; async function init() { const video = document.getElementById('video'); const ui = video['ui']; const player = ui.getControls().getPlayer(); const listDiv = document.getElementById('list'); player.configure({ manifest: { dash: { ignoreDrmInfo: true } }, streaming: { crossOrigin: 'anonymous' } // Forza il browser a non inviare credenziali }); player.getNetworkingEngine().registerRequestFilter((type, request) => { if (request.uris[0].includes('pluscdn.pl')) { // Proviamo a forzare una richiesta ultra-base request.allowCrossOrigin = true; request.headers['User-Agent'] = 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/123.0.0.0 Safari/537.36'; // Tentativo disperato di svuotare tutto quello che Chrome ci permette delete request.headers['Origin']; delete request.headers['Referer']; } }); try { const res = await fetch(SOURCE); const text = await res.text(); const lines = text.split('\n'); let ch = { drm: {} }; for (let line of lines) { line = line.trim(); if (line.includes('license_key=')) { const m = line.match(/[a-f0-9]{32}/gi); if (m) ch.drm[m[0]] = m[1]; } else if (line.startsWith('#EXTINF:')) { ch.logo = (line.match(/tvg-logo="([^"]+)"/i) || [])[1] || ""; ch.group = (line.match(/group-title="([^"]+)"/i) || [])[1] || "ALTRI"; ch.name = (line.match(/,(.*)/) || [])[1] || "Canale"; } else if (line.startsWith('http')) { const url = line; const currentCh = {...ch}; let grid = document.getElementById(currentCh.group); if (!grid) { const title = document.createElement('div'); title.className = 'group-title'; title.innerText = currentCh.group; listDiv.appendChild(title); grid = document.createElement('div'); grid.className = 'grid'; grid.id = currentCh.group; listDiv.appendChild(grid); } const card = document.createElement('div'); card.className = 'card'; card.innerHTML = `
${currentCh.name}
`; card.onclick = async () => { await player.unload(); player.configure({ drm: { clearKeys: currentCh.drm } }); // Se non va in HTTPS, il browser su GitHub lo blocca a priori (Mixed Content) await player.load(url.replace('http://', 'https://')); }; grid.appendChild(card); ch = { drm: {} }; } } } catch(e) {} } document.addEventListener('shaka-ui-loaded', init);
